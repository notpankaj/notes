2. ACTIONS

const increment = () => { 
    return {
        type : "INCREMENT",
        payload : 2,
    } 
};
const decrement = () => ({ type : "DECREMENT" });


-----------------
3. REDUCER -> updates the store based on action
-----------------
const counter = (state = 0, action) => {
  switch(action.type){
    case 'INCREMENT':
      return state + action.payload;
    case 'DECREMENT':
        return state - 1;
    }
}
    
-----------------
1. STORE -> GLOBALIZED STATE
